1.1 Land Query 

 

SELECT p1.[prop_id] 

 ,p1.[geo_id] 

 ,pv1.[rgn_cd] 

 ,pv1.[last_appraiser_id] 

 ,pv1.[prop_val_yr] 

 ,pv1.[property_use_cd] 

 ,ld1.[land_type_cd] 

 ,ld1.[mkt_flat_val] 

 ,ld1.[mkt_calc_val] 

 ,ld1.[mkt_adj_val] 

 ,ld1.[size_square_feet] 

 ,p1.[dba_name] 

FROM dbo.land_detail AS ld1 WITH (NOLOCK) 

JOIN dbo.prop_supp_assoc AS psa1 WITH (NOLOCK) ON 
psa1.[prop_id] = ld1.[prop_id] 

 AND psa1.[owner_tax_yr] = ld1.[prop_val_yr] 

 AND psa1.[sup_num] = ld1.[sup_num] 

JOIN dbo.property AS p1 WITH (NOLOCK) ON p1.[prop_id] = 
psa1.[prop_id] 

JOIN dbo.property_val AS pv1 WITH (NOLOCK) ON pv1.[prop_id] 
= ld1.[prop_id] 

 AND pv1.[prop_val_yr] = ld1.[prop_val_yr] 

 AND pv1.[sup_num] = ld1.[sup_num] 

INNER JOIN pacs_system ON pv1.prop_val_yr = 
pacs_system.appr_yr 

WHERE pv1.[last_appraiser_id] = 643 

 AND pv1.[prop_inactive_dt] IS NULL

 AND 4000

1.4 Condo Query 

 

SELECT property.dba_name 

 ,property_val.legal_desc 

 ,property_val.prop_id 

 ,appraiser.appraiser_full_name 

 ,property.geo_id 

 ,property_val.land_hstd_val 

 ,property_val.land_non_hstd_val 

 ,property_val.imprv_hstd_val 

 ,property_val.imprv_non_hstd_val 

 ,imprv_detail.imprv_det_type_cd 

 ,imprv.adjusted_val 

 ,imprv.flat_val 

 ,property_profile.living_area 

 ,imprv_detail.imprv_det_class_cd 

 ,imprv_detail.condition_cd 

 ,imprv_detail.yr_built 

 ,imprv.effective_yr_blt 

FROM ( 

 ( 

 ( 

 ( 

 ( 

 ( 

 property_val INNER JOIN property ON 
property_val.prop_id = property.prop_id 

 ) INNER JOIN OWNER ON 
(property_val.prop_val_yr = OWNER.owner_tax_yr) 

 AND (property_val.prop_id = 
OWNER.prop_id) 

 ) INNER JOIN pacs_system ON 
property_val.prop_val_yr = pacs_system.appr_yr 

 ) INNER JOIN appraiser ON 
property_val.last_appraiser_id = appraiser.appraiser_id 

 ) INNER JOIN imprv ON (property_val.sup_num = 
imprv.sup_num) 

 AND (property_val.prop_val_yr = 
imprv.prop_val_yr) 

 AND (property_val.prop_id = imprv.prop_id) 

 ) INNER JOIN property_profile ON 
(property_val.sup_num = property_profile.sup_num) 

 AND (property_val.prop_val_yr = 
property_profile.prop_val_yr) 

 AND (property_val.prop_id = 
property_profile.prop_id) 

 ) 

INNER JOIN imprv_detail ON (imprv.sale_id = 
imprv_detail.sale_id) 

 AND (property_val.sup_num = imprv_detail.sup_num) 

 AND (imprv.imprv_id = imprv_detail.imprv_id) 

 AND (property_val.prop_val_yr = 
imprv_detail.prop_val_yr) 

 AND (property_val.prop_id = imprv_detail.prop_id) 

WHERE ( 

 ( 

 (property.dba_name) LIKE '%condo%' 

 OR [property_val].[legal_desc] LIKE '%condo%' 

 ) 

 AND ((property_val.sup_num) = 0) 

 AND ((property_val.prop_inactive_dt) IS NULL) 

 AND ((imprv.sale_id) = 0) 

 AND ((imprv_detail.imprv_det_type_cd) = 'bas') 

 ); 

 


1.5 Commercial Income And Expense 
Properties 

Used by the Commercial Appraiser during valuations 
review. The query pulls only those properties that 
have had an Income Valuation created. If the 
property has been converted to Marshall & Swift the 
Occupancy Type(s) are provided. 

SELECT DISTINCT pv.prop_id 

 ,pv.prop_val_yr 

 ,p.geo_id 

 ,ac.file_as_name 

 ,cmso.occupancy_code AS 'type' 

 ,cmso.occupancy_description 

 ,inc.income_id 

 ,inc.econ_area 

 ,inc.prop_type_cd 

 ,'$' + convert(VARCHAR(12), ipa.income_value, 1) AS 
income_value 

FROM property_val pv 

JOIN income_prop_assoc ipa ON pv.prop_id = ipa.prop_id 

 AND pv.prop_val_yr = ipa.prop_val_yr 

 AND pv.sup_num = ipa.sup_num 

JOIN property p ON pv.prop_id = p.prop_id 

JOIN OWNER AS o ON ipa.prop_id = o.prop_id 

 AND ipa.prop_val_yr = o.owner_tax_yr 

 AND ipa.sup_num = o.sup_num 

JOIN account AS ac ON o.owner_id = ac.acct_id 

LEFT JOIN imprv_detail_cms_occupancy AS cmso ON ipa.prop_id 
= cmso.prop_id 

 AND ipa.prop_val_yr = cmso.prop_val_yr 

 AND ipa.sup_num = cmso.sup_num 

 AND cmso.sale_id = 0 

JOIN income AS inc ON ipa.income_id = inc.income_id 

 AND ipa.prop_val_yr = inc.income_yr 

 AND ipa.sup_num = inc.sup_num 

INNER JOIN pacs_system ON pv.prop_val_yr = 
pacs_system.appr_yr 

WHERE pv.last_appraiser_id = 643 

 AND pv.prop_inactive_dt IS NULL 

ORDER BY pv.prop_id 

 ,pv.prop_val_yr

1.7 Commercial Imprv Value Change From 
Prior Year – BAS Only 

Used by the Commercial Appraiser (Harold) to 
compare Improvement value changes from the prior 
year. Very similar to the previous query—but this 
pulls only the ‘BAS’ improvement detail. 

SELECT DISTINCT pv.prop_id 

 ,p.geo_id 

 ,ac.file_as_name 

 ,id.imprv_id 

 , 

 --id.imprv_det_id, 

 id.imprv_det_val 

 ,pv.prop_val_yr 

FROM property_val pv 

JOIN prop_supp_assoc psa ON pv.prop_id = psa.prop_id 

 AND pv.prop_val_yr = psa.owner_tax_yr 

 AND pv.sup_num = psa.sup_num 

JOIN property p ON pv.prop_id = p.prop_id 

JOIN OWNER AS o ON psa.prop_id = o.prop_id 

 AND psa.owner_tax_yr = o.owner_tax_yr 

 AND psa.sup_num = o.sup_num 

JOIN account AS ac ON o.owner_id = ac.acct_id 

JOIN imprv AS i ON psa.prop_id = i.prop_id 

 AND psa.owner_tax_yr = i.prop_val_yr 

 AND psa.sup_num = i.sup_num 

 AND i.sale_id = 0 

JOIN imprv_detail AS id ON i.prop_id = id.prop_id 

 AND i.prop_val_yr = id.prop_val_yr 

 AND i.sup_num = id.sup_num 

 AND i.sale_id = id.sale_id 

WHERE pv.prop_val_yr IN ( 

 '2018' 

 ,'2019' 

 ) 

 AND pv.last_appraiser_id = 643 

 AND pv.prop_inactive_dt IS NULL 

 AND id.imprv_det_type_cd = 'bas' 

ORDER BY pv.prop_id
